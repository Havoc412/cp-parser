int main() {
    int a = 10;
    int b = -20;
    
    // 传统 if-else 结构
    else {
        b = b - 1;
    }
    
    // 新增的 if-then 结构
    if (a == b) then {
        a = a * 2;
    }
    
    // if-then-else 结构
    if (a && b) then {
        a = a | b;
    } else {
        b = b + 1;
    }

    
    return 0;
}
